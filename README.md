# TO RUN APPLICATION 
- clone to local machine
- cd server
- run "npm install"
- create a dot env file with the following environment variables:
SESSION_SECRET=
PORT=
DB_NAME=
JWT_KEY=
- assign values to the variables
- cd .././client
- run "npm install"
- cd .././server
- run "npm run-script buildrun"
